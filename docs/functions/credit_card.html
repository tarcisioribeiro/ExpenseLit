491    def card_remaining_limit(self, selected_card: str):
492        """
493        Calcula o limite restante do cartão de crédito.
494
495        Parameters
496        ----------
497        selected_card : str
498            O cartão de crédito selecionado.
499
500        Returns
501        -------
502        remaining_limit : float
503            0 valor do limite restante do cartão.
504        """
505        card_total_limit = self.card_limit(selected_card)
506        not_payed_expenses = self.not_payed_expenses(selected_card)
507        month_card_expenses = self.month_expenses(selected_card, actual_year, string_actual_month)
508        future_card_expenses = self.future_expenses(selected_card)
509        total_card_expenses = month_card_expenses + \
510            future_card_expenses + not_payed_expenses
511        remaining_limit = card_total_limit - total_card_expenses
512        return remaining_limit
