 23    def mount_card_graph(self, selected_card: str):
 24        """
 25        Elabora o gráfico de despesas de cartão de crédito.
 26        """
 27        credit_card = Credit_Card()
 28        int_actual_year = int(actual_year)
 29
 30        not_payed_expenses = credit_card.not_payed_expenses(selected_card)
 31        month_expenses = credit_card.month_expenses(selected_card, int_actual_year, string_actual_month)
 32        future_expenses = credit_card.future_expenses(selected_card)
 33        card_limit = credit_card.card_limit(selected_card)
 34        remaining_limit = credit_card.card_remaining_limit(selected_card)
 35
 36        st.info(body="Cartão {}".format(selected_card))
 37        credit_card_data_df = pd.DataFrame(
 38            {
 39                "Categoria": ["Faturas não pagas", "Mês Atual", "Futuro", "Limite Restante"],
 40                "Valor": [not_payed_expenses, month_expenses, future_expenses, remaining_limit],
 41                "Porcentagem": [
 42                    (not_payed_expenses / card_limit) *
 43                    100 if card_limit != 0 else 0,
 44                    (month_expenses / card_limit) *
 45                    100 if card_limit != 0 else 0,
 46                    (future_expenses / card_limit) *
 47                    100 if card_limit != 0 else 0,
 48                    (remaining_limit / card_limit) *
 49                    100 if card_limit != 0 else 0,
 50                ],
 51            }
 52        )
 53        credit_card_data_df["Valor"] = credit_card_data_df["Valor"].apply(lambda x: f"R$ {x:.2f}".replace(".", ","))
 54        credit_card_data_df["Porcentagem"] = credit_card_data_df["Porcentagem"].apply(lambda x: f"{x:.2f}%".replace(".", ","))
 55        st.dataframe(credit_card_data_df, hide_index=True, use_container_width=True)

 57    def mount_balance(self):
 58        """
 59        Elabora e estrutura o balaço geral das contas do usuário.
 60
 61        Returns

 63        values_list : list
 64            A lista com os valores de despesas e receitas.

 66        balance = GetBalance().balance()
 67
 68        user_name, user_document = Login().get_user_data(return_option="user_doc_name")
 69
 70        values_list = []
 71
 72        final_str_balance = Variable().treat_complex_string(balance)

 74
 75        ticket_revenue_ammount = QueryExecutor().simple_consult_query(query=ticket_revenue_query, params=(user_name, user_document))

 77        ticket_revenue_ammount = float(ticket_revenue_ammount)
 78        ticket_expense_ammount = QueryExecutor().simple_consult_query(query=ticket_expense_query, params=(user_name, user_document))

 80        ticket_expense_ammount = float(ticket_expense_ammount)
 81        ticket_ammount = ticket_revenue_ammount - ticket_expense_ammount
 82        final_str_ticket_ammount = Variable().treat_complex_string(ticket_ammount)
 83        values_list.append(final_str_ticket_ammount)

 87        fund_revenue_ammount = float(fund_revenue_ammount)

 90        fund_expense_ammount = float(fund_expense_ammount)
 91        fund_ammount = fund_revenue_ammount - fund_expense_ammount
 92        final_str_fund_ammount = Variable().treat_complex_string(fund_ammount)
 93        values_list.append(final_str_fund_ammount)
 94
 95        loan_ammount = QueryExecutor().simple_consult_query(query=loan_expense_query, params=(user_name, user_document))
 96        loan_ammount = QueryExecutor().treat_simple_result(loan_ammount, to_remove_list)
 97        final_str_loan_ammount = Variable().treat_complex_string(loan_ammount)
 98        values_list.append(final_str_loan_ammount)
 99
100        debts_ammount = QueryExecutor().simple_consult_query(query=debts_expense_query, params=(user_name, user_document))
101        debts_ammount = QueryExecutor().treat_simple_result(debts_ammount, to_remove_list)
102        final_str_debts_ammount = Variable().treat_complex_string(debts_ammount)
103        values_list.append(final_str_debts_ammount)
